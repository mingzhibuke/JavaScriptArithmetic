时间效率 O(n*n) , 算法参考自三数之后为定值
```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var threeSumClosest = function(nums, target) {
    nums.sort( (a, b) => a - b ); 

    let sum = 2 << 30; 

    for(let i = 0; i < nums.length - 2; i++){
        let low = i + 1, 
            high = nums.length - 1; 

        while(low < high){
            let termSum = nums[low] + nums[high] + nums[i], 
                diff = termSum - target; 

            if(Math.abs(diff) < Math.abs(sum - target))
                sum = termSum; 

            if(diff === 0)
                return sum; 
            else if (diff < 0)
                low ++; 
            else 
                high --; 
        }
    }
    return sum; 
};

let nums = [-1, 2, 1, -4], 
    target = 1; 

console.log(threeSumClosest(nums, target)); 
```
